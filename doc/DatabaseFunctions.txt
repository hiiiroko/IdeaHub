add_comment
在指定视频下创建一条新评论。
let { data, error } = await supabase
  .rpc('add_comment', {
    p_content, 
    p_parent_comment_id, 
    target_video_id
  })
if (error) console.error(error)
else console.log(data)

get_daily_likes_trend
返回当前登录用户视频的“每日点赞趋势”数据，用于绘制折线图 / 柱状图。
let { data, error } = await supabase
  .rpc('get_daily_likes_trend')
if (error) console.error(error)
else console.log(data)

get_interaction_stats
返回当前登录用户所有视频的整体互动统计数据，用于仪表盘展示。
let { data, error } = await supabase
  .rpc('get_interaction_stats')

else console.log(data)

increment_view_count

let { data, error } = await supabase

if (error) console.error(error)
else console.log(data)
